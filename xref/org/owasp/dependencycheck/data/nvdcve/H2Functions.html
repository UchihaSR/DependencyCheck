39      /**
40       * Adds a CPE to a vulnerability; if the CPE is not contained in the
41       * database it is first added.
42       *
43       * @param conn the database connection
44       * @param vulnerabilityId the vulnerability id
45       * @param part the CPE part
46       * @param vendor the CPE vendor
47       * @param product the CPE product
48       * @param version the CPE version
49       * @param update the CPE update version
50       * @param edition the CPE edition
51       * @param language the CPE language
52       * @param swEdition the CPE software edition
53       * @param targetSw the CPE target software
54       * @param targetHw the CPE target hardware
55       * @param other the CPE other
56       * @param ecosystem the ecosystem
57       * @param versionEndExcluding a version range to identify the software
58       * @param versionEndIncluding a version range to identify the software
59       * @param versionStartExcluding a version range to identify the software
60       * @param versionStartIncluding a version range to identify the software
61       * @param vulnerable a flag indicating whether or not the software is
62       * vulnerable
63       * @throws SQLException thrown if there is an error adding the CPE or
64       * software reference
65       */
66      public static void insertSoftware(final Connection conn, int vulnerabilityId, String part, String vendor,
67              String product, String version, String update, String edition, String language, String swEdition,
68              String targetSw, String targetHw, String other, String ecosystem, String versionEndExcluding,
69              String versionEndIncluding, String versionStartExcluding, String versionStartIncluding, Boolean vulnerable) throws SQLException {
70          int cpeID = 0;
71          final PreparedStatement selectCpeId = conn.prepareStatement("SELECT id, ecosystem FROM cpeEntry WHERE part=? AND vendor=? AND product=? "
72                  + "AND version=? AND update_version=? AND edition=? AND lang=? AND sw_edition=? AND target_sw=? AND target_hw=? AND other=?");
73          selectCpeId.setString(1, part);
74          selectCpeId.setString(2, vendor);
75          selectCpeId.setString(3, product);
76          selectCpeId.setString(4, version);
77          selectCpeId.setString(5, update);
78          selectCpeId.setString(6, edition);
79          selectCpeId.setString(7, language);
80          selectCpeId.setString(8, swEdition);
81          selectCpeId.setString(9, targetSw);
82          selectCpeId.setString(10, targetHw);
83          selectCpeId.setString(11, other);
84  
85          try (ResultSet rs = selectCpeId.executeQuery()) {
86              if (rs.next()) {
87                  cpeID = rs.getInt(1);
88                  final String e = rs.getString(2);
89                  if (e == null && ecosystem != null) {
90                      final PreparedStatement updateEcosystem = conn.prepareStatement("UPDATE cpeEntry SET ecosystem=? WHERE id=?");
91                      updateEcosystem.setString(1, ecosystem);
92                      updateEcosystem.setInt(2, cpeID);
93                  }
94              }
95          }
96          if (cpeID == 0) {
97              final String[] returnedColumns = {"id"};
98              final PreparedStatement insertCpe = conn.prepareStatement("INSERT INTO cpeEntry (part, vendor, product, version, update_version, "
99                      + "edition, lang, sw_edition, target_sw, target_hw, other, ecosystem) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",
100                     returnedColumns);
101             insertCpe.setString(1, part);
102             insertCpe.setString(2, vendor);
103             insertCpe.setString(3, product);
104             insertCpe.setString(4, version);
105             insertCpe.setString(5, update);
106             insertCpe.setString(6, edition);
107             insertCpe.setString(7, language);
108             insertCpe.setString(8, swEdition);
109             insertCpe.setString(9, targetSw);
110             insertCpe.setString(10, targetHw);
111             insertCpe.setString(11, other);
112             setStringOrNull(insertCpe, 12, ecosystem);
113             insertCpe.executeUpdate();
114             try (ResultSet rs = insertCpe.getGeneratedKeys()) {
115                 if (rs.next()) {
116                     cpeID = rs.getInt(1);
117                 }
118             }
119         }
120         //CSON: ParameterNumber
121 
122         final PreparedStatement insertSoftware = conn.prepareStatement("INSERT INTO software (cveid, cpeEntryId, "
123                 + "versionEndExcluding, versionEndIncluding, versionStartExcluding, versionStartIncluding, "
124                 + "vulnerable) VALUES (?, ?, ?, ?, ?, ?, ?)");
125         insertSoftware.setInt(1, vulnerabilityId);
126         insertSoftware.setInt(2, cpeID);
127 
128         setStringOrNull(insertSoftware, 3, versionEndExcluding);
129         setStringOrNull(insertSoftware, 4, versionEndIncluding);
130         setStringOrNull(insertSoftware, 5, versionStartExcluding);
131         setStringOrNull(insertSoftware, 6, versionStartIncluding);
132         setBooleanOrNull(insertSoftware, 7, vulnerable);
133         insertSoftware.execute();
134     }

137     /**
138      * Updates or inserts the vulnerability into the database. If updating a
139      * vulnerability the method will delete all software, CWE, and references
140      * and new entries will be added later.
141      *
142      * @param conn the database connection
143      * @param cve the CVE identifier
144      * @param description the vulnerability description
145      * @param v2Severity the CVSS v2 severity
146      * @param v2ExploitabilityScore the CVSS v2 exploitability score
147      * @param v2ImpactScore the CVSS v2 impact score
148      * @param v2AcInsufInfo the CVSS v2 AcInsufInfo
149      * @param v2ObtainAllPrivilege the CVSS v2 obtain all privilege flag
150      * @param v2ObtainUserPrivilege the CVSS v2 obtain user privilege flag
151      * @param v2ObtainOtherPrivilege the CVSS v2 obtain other privilege flag
152      * @param v2UserInteractionRequired the CVSS v2 user interaction required
153      * flag
154      * @param v2Score the CVSS v2 score
155      * @param v2AccessVector the CVSS v2 access vector
156      * @param v2AccessComplexity the CVSS v2 access complexity
157      * @param v2Authentication the CVSS v2 authentication
158      * @param v2ConfidentialityImpact the CVSS v2 confidentiality impact
159      * @param v2IntegrityImpact the CVSS v2 integrity impact
160      * @param v2AvailabilityImpact the CVSS v2 availability impact
161      * @param v2Version the CVSS v2 version
162      * @param v3ExploitabilityScore the CVSS v3 exploitability score
163      * @param v3ImpactScore the CVSS v3 impact score
164      * @param v3AttackVector the CVSS v3 attack vector
165      * @param v3AttackComplexity the CVSS v3 attack complexity
166      * @param v3PrivilegesRequired the CVSS v3 privilege required flag
167      * @param v3UserInteraction the CVSS v3 user interaction required flag
168      * @param v3Scope the CVSS v3 scope
169      * @param v3ConfidentialityImpact the CVSS v3 confidentiality impact
170      * @param v3IntegrityImpact the CVSS v3 integrity impact
171      * @param v3AvailabilityImpact the CVSS v3 availability impact
172      * @param v3BaseScore the CVSS v3 base score
173      * @param v3BaseSeverity the CVSS v3 base severity
174      * @param v3Version the CVSS v3 version
175      * @return a result set containing the vulnerability id
176      * @throws SQLException thrown if there is an error updating or inserting
177      * the vulnerability
178      */
179     public static ResultSet updateVulnerability(final Connection conn, String cve,
180             String description, String v2Severity, Float v2ExploitabilityScore,
181             Float v2ImpactScore, Boolean v2AcInsufInfo, Boolean v2ObtainAllPrivilege,
182             Boolean v2ObtainUserPrivilege, Boolean v2ObtainOtherPrivilege, Boolean v2UserInteractionRequired,
183             Float v2Score, String v2AccessVector, String v2AccessComplexity,
184             String v2Authentication, String v2ConfidentialityImpact, String v2IntegrityImpact,
185             String v2AvailabilityImpact, String v2Version, Float v3ExploitabilityScore,
186             Float v3ImpactScore, String v3AttackVector, String v3AttackComplexity,
187             String v3PrivilegesRequired, String v3UserInteraction, String v3Scope,
188             String v3ConfidentialityImpact, String v3IntegrityImpact, String v3AvailabilityImpact,
189             Float v3BaseScore, String v3BaseSeverity, String v3Version) throws SQLException {
190 
191         final SimpleResultSet ret = new SimpleResultSet();
192         ret.addColumn("id", Types.INTEGER, 10, 0);
193 
194         int vulnerabilityId = 0;
195         final PreparedStatement selectVulnerabilityId = conn.prepareStatement("SELECT id FROM VULNERABILITY CVE WHERE cve=?");
196         selectVulnerabilityId.setString(1, cve);
197         try (ResultSet rs = selectVulnerabilityId.executeQuery()) {
198             if (rs.next()) {
199                 vulnerabilityId = rs.getInt(1);
200             }
201         }
202 
203         final PreparedStatement merge;
204         if (vulnerabilityId > 0) {
205             //do deletes and updates
206             try (PreparedStatement refs = conn.prepareStatement("DELETE FROM reference WHERE cveid = ?")) {
207                 refs.setInt(1, vulnerabilityId);
208                 refs.executeUpdate();
209             }
210             try (PreparedStatement software = conn.prepareStatement("DELETE FROM software WHERE cveid = ?")) {
211                 software.setInt(1, vulnerabilityId);
212                 software.executeUpdate();
213             }
214             try (PreparedStatement cwe = conn.prepareStatement("DELETE FROM cweEntry WHERE cveid = ?")) {
215                 cwe.setInt(1, vulnerabilityId);
216                 cwe.executeUpdate();
217             }
218             merge = conn.prepareStatement("UPDATE VULNERABILITY SET description=?, "
219                     + "v2Severity=?, v2ExploitabilityScore=?, "
220                     + "v2ImpactScore=?, v2AcInsufInfo=?, v2ObtainAllPrivilege=?, "
221                     + "v2ObtainUserPrivilege=?, v2ObtainOtherPrivilege=?, v2UserInteractionRequired=?, "
222                     + "v2Score=?, v2AccessVector=?, v2AccessComplexity=?, "
223                     + "v2Authentication=?, v2ConfidentialityImpact=?, v2IntegrityImpact=?, "
224                     + "v2AvailabilityImpact=?, v2Version=?, v3ExploitabilityScore=?, "
225                     + "v3ImpactScore=?, v3AttackVector=?, v3AttackComplexity=?, "
226                     + "v3PrivilegesRequired=?, v3UserInteraction=?, v3Scope=?, "
227                     + "v3ConfidentialityImpact=?, v3IntegrityImpact=?, v3AvailabilityImpact=?, "
228                     + "v3BaseScore=?, v3BaseSeverity=?, v3Version=? "
229                     + "WHERE id=?");
230         } else {
231             //just do insert
232             final String[] returnedColumns = {"id"};
233             merge = conn.prepareStatement("INSERT INTO VULNERABILITY (description, "
234                     + "v2Severity, v2ExploitabilityScore, "
235                     + "v2ImpactScore, v2AcInsufInfo, v2ObtainAllPrivilege, "
236                     + "v2ObtainUserPrivilege, v2ObtainOtherPrivilege, v2UserInteractionRequired, "
237                     + "v2Score, v2AccessVector, v2AccessComplexity, "
238                     + "v2Authentication, v2ConfidentialityImpact, v2IntegrityImpact, "
239                     + "v2AvailabilityImpact, v2Version, v3ExploitabilityScore, "
240                     + "v3ImpactScore, v3AttackVector, v3AttackComplexity, "
241                     + "v3PrivilegesRequired, v3UserInteraction, v3Scope, "
242                     + "v3ConfidentialityImpact, v3IntegrityImpact, v3AvailabilityImpact, "
243                     + "v3BaseScore, v3BaseSeverity, v3Version, cve) VALUES (?, ?, ?, ?, ?, ?, "
244                     + "?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",
245                     returnedColumns);
246         }
247 
248         merge.setString(1, description);
249 
250         setStringOrNull(merge, 2, v2Severity);
251         setFloatOrNull(merge, 3, v2ExploitabilityScore);
252         setFloatOrNull(merge, 4, v2ImpactScore);
253         setBooleanOrNull(merge, 5, v2AcInsufInfo);
254         setBooleanOrNull(merge, 6, v2ObtainAllPrivilege);
255         setBooleanOrNull(merge, 7, v2ObtainUserPrivilege);
256         setBooleanOrNull(merge, 8, v2ObtainOtherPrivilege);
257         setBooleanOrNull(merge, 9, v2UserInteractionRequired);
258         setFloatOrNull(merge, 10, v2Score);
259         setStringOrNull(merge, 11, v2AccessVector);
260         setStringOrNull(merge, 12, v2AccessComplexity);
261         setStringOrNull(merge, 13, v2Authentication);
262         setStringOrNull(merge, 14, v2ConfidentialityImpact);
263         setStringOrNull(merge, 15, v2IntegrityImpact);
264         setStringOrNull(merge, 16, v2AvailabilityImpact);
265         setStringOrNull(merge, 17, v2Version);
266         setFloatOrNull(merge, 18, v3ExploitabilityScore);
267         setFloatOrNull(merge, 19, v3ImpactScore);
268         setStringOrNull(merge, 20, v3AttackVector);
269         setStringOrNull(merge, 21, v3AttackComplexity);
270         setStringOrNull(merge, 22, v3PrivilegesRequired);
271         setStringOrNull(merge, 23, v3UserInteraction);
272         setStringOrNull(merge, 24, v3Scope);
273         setStringOrNull(merge, 25, v3ConfidentialityImpact);
274         setStringOrNull(merge, 26, v3IntegrityImpact);
275         setStringOrNull(merge, 27, v3AvailabilityImpact);
276         setFloatOrNull(merge, 28, v3BaseScore);
277         setStringOrNull(merge, 29, v3BaseSeverity);
278         setStringOrNull(merge, 30, v3Version);
279 
280         if (vulnerabilityId == 0) {
281             merge.setString(31, cve);
282         } else {
283             merge.setInt(31, vulnerabilityId);
284         }
285         final int count = merge.executeUpdate();
286         if (vulnerabilityId == 0) {
287             try (ResultSet rs = merge.getGeneratedKeys()) {
288                 if (rs.next()) {
289                     vulnerabilityId = rs.getInt(1);
290                 }
291             }
292         }
293 
294         ret.addRow(vulnerabilityId);
295         return ret;
296     }

299     /**
300      * Sets a parameter value on a prepared statement with null checks.
301      *
302      * @param ps the prepared statement
303      * @param i the parameter index
304      * @param value the value
305      * @throws SQLException thrown if there is an error setting the parameter
306      */
307     private static void setStringOrNull(PreparedStatement ps, int i, String value) throws SQLException {
308         if (value == null) {
309             ps.setNull(i, java.sql.Types.NULL);
310         } else {
311             ps.setString(i, value);
312         }
313     }
