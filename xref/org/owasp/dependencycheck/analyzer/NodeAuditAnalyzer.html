110     /**
111      * Initializes the analyzer once before any analysis is performed.
112      *
113      * @param engine a reference to the dependency-check engine
114      * @throws InitializationException if there's an error during initialization
115      */
116     @Override
117     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
118         if (!isEnabled() || !getFilesMatched()) {
119             this.setEnabled(false);
120             return;
121         }
122         if (searcher == null) {
123             LOGGER.debug("Initializing {}", getName());
124             try {
125                 searcher = new NodeAuditSearch(getSettings());
126             } catch (MalformedURLException ex) {
127                 setEnabled(false);
128                 throw new InitializationException("The configured URL to NPM Audit API is malformed", ex);
129             }
130             try {
131                 final Settings settings = engine.getSettings();
132                 final boolean nodeEnabled = settings.getBoolean(Settings.KEYS.ANALYZER_NODE_PACKAGE_ENABLED);
133                 if (!nodeEnabled) {
134                     LOGGER.warn("The Node Package Analyzer has been disabled; the resulting report will only "
135                             + "contain the known vulnerable dependency - not a bill of materials for the node project.");
136                 }
137             } catch (InvalidSettingException ex) {
138                 throw new InitializationException("Unable to read configuration settings", ex);
139             }
140         }
141     }

174     @Override
175     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
176         if (dependency.getDisplayFileName().equals(dependency.getFileName())) {
177             engine.removeDependency(dependency);
178         }
179         final File packageLock = dependency.getActualFile();
180         if (!packageLock.isFile() || packageLock.length() == 0 || !shouldProcess(packageLock)) {
181             return;
182         }
183         final File packageJson = new File(packageLock.getParentFile(), "package.json");
184         final List<Advisory> advisories;
185         final Map<String, String> dependencyMap = new HashMap<>();
186         if (packageJson.isFile()) {
187             advisories = analyzePackage(packageLock, packageJson, dependency, dependencyMap);
188         } else {
189             advisories = legacyAnalysis(packageLock, dependency, dependencyMap);
190         }
191         try {
192             processResults(advisories, engine, dependency, dependencyMap);
193         } catch (CpeValidationException ex) {
194             throw new UnexpectedAnalysisException(ex);
195         }
196     }

198     /**
199      * Processes the advisories creating the appropriate dependency objects and
200      * adding the resulting vulnerabilities.
201      *
202      * @param advisories a collection of advisories from npm
203      * @param engine a reference to the analysis engine
204      * @param dependency a reference to the package-lock.json dependency
205      * @param dependencyMap a collection of module/version pairs obtained from
206      * the package-lock file - used in case the advisories do not include a
207      * version number
208      * @throws CpeValidationException thrown when a CPE cannot be created
209      */
210     private void processResults(final List<Advisory> advisories, Engine engine,
211             Dependency dependency, Map<String, String> dependencyMap)
212             throws CpeValidationException {
213         for (Advisory advisory : advisories) {
214             //Create a new vulnerability out of the advisory returned by nsp.
215             final Vulnerability vuln = new Vulnerability();
216             vuln.setDescription(advisory.getOverview());
217             vuln.setName(String.valueOf(advisory.getId()));
218             vuln.setUnscoredSeverity(advisory.getSeverity());
219             vuln.setSource(Vulnerability.Source.NPM);
220             vuln.addReference(
221                     "Advisory " + advisory.getId() + ": " + advisory.getTitle(),
222                     advisory.getReferences(),
223                     null
224             );
225 
226             //Create a single vulnerable software object - these do not use CPEs unlike the NVD.
227             final VulnerableSoftwareBuilder builder = new VulnerableSoftwareBuilder();
228             builder.part(Part.APPLICATION).product(advisory.getModuleName().replace(" ", "_"))
229                     .version(advisory.getVulnerableVersions().replace(" ", ""));
230             final VulnerableSoftware vs = builder.build();
231             vuln.setVulnerableSoftware(Collections.singleton(vs));
232 
233             String version = advisory.getVersion();
234             if (version == null && dependencyMap.containsKey(advisory.getModuleName())) {
235                 version = dependencyMap.get(advisory.getModuleName());
236             }
237 
238             final Dependency existing = findDependency(engine, advisory.getModuleName(), version);
239             if (existing == null) {
240                 final Dependency nodeModule = createDependency(dependency, advisory.getModuleName(), version, "transitive");
241                 nodeModule.addVulnerability(vuln);
242                 engine.addDependency(nodeModule);
243             } else {
244                 replaceOrAddVulnerability(existing, vuln);
245             }
246         }
247     }

249     /**
250      * Evaluates if the vulnerability is already present; if it is the
251      * vulnerability is not added.
252      *
253      * @param dependency a reference to the dependency being analyzed
254      * @param vuln the vulnerability to add
255      */
256     private void replaceOrAddVulnerability(Dependency dependency, Vulnerability vuln) {
257         boolean found = false;
258         for (Vulnerability existing : dependency.getVulnerabilities()) {
259             for (Reference ref : existing.getReferences()) {
260                 if (ref.getName() != null
261                         && vuln.getSource().toString().equals("NPM")
262                         && ref.getName().equals("https://nodesecurity.io/advisories/" + vuln.getName())) {
263                     found = true;
264                 }
265             }
266         }
267         if (!found) {
268             dependency.addVulnerability(vuln);
269         }
270     }

272     /**
273      * Analyzes the package and package-lock files by extracting dependency
274      * information, creating a payload to submit to the npm audit API,
275      * submitting the payload, and returning the identified advisories.
276      *
277      * @param lockFile a reference to the package-lock.json
278      * @param packageFile a reference to the package.json
279      * @param dependency a reference to the dependency-object for the
280      * package-lock.json
281      * @param dependencyMap a collection of module/version pairs; during
282      * creation of the payload the dependency map is populated with the
283      * module/version information.
284      * @return a list of advisories
285      * @throws AnalysisException thrown when there is an error creating or
286      * submitting the npm audit API payload
287      */
288     private List<Advisory> analyzePackage(final File lockFile, final File packageFile,
289             Dependency dependency, Map<String, String> dependencyMap)
290             throws AnalysisException {
291         try (JsonReader lockReader = Json.createReader(FileUtils.openInputStream(lockFile));
292                 JsonReader packageReader = Json.createReader(FileUtils.openInputStream(packageFile))) {
293 
294             // Retrieves the contents of package-lock.json from the Dependency
295             final JsonObject lockJson = lockReader.readObject();
296             // Retrieves the contents of package-lock.json from the Dependency
297             final JsonObject packageJson = packageReader.readObject();
298 
299             // Modify the payload to meet the NPM Audit API requirements
300             final JsonObject payload = NpmPayloadBuilder.build(lockJson, packageJson, dependencyMap,
301                     getSettings().getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_SKIPDEV, false));
302 
303             // Submits the package payload to the nsp check service
304             return searcher.submitPackage(payload);
305 
306         } catch (URLConnectionFailureException e) {
307             this.setEnabled(false);
308             throw new AnalysisException("Failed to connect to the NPM Audit API (NodeAuditAnalyzer); the analyzer "
309                     + "is being disabled and may result in false negatives.", e);
310         } catch (IOException e) {
311             LOGGER.debug("Error reading dependency or connecting to NPM Audit API", e);
312             this.setEnabled(false);
313             throw new AnalysisException("Failed to read results from the NPM Audit API (NodeAuditAnalyzer); "
314                     + "the analyzer is being disabled and may result in false negatives.", e);
315         } catch (JsonException e) {
316             throw new AnalysisException(String.format("Failed to parse %s file from the NPM Audit API "
317                     + "(NodeAuditAnalyzer).", lockFile.getPath()), e);
318         } catch (SearchException ex) {
319             final File yarnCheck = new File(lockFile.getParentFile(), "yarn.lock");
320             if (yarnCheck.exists()) {
321                 final String msg = "NodeAuditAnalyzer filed on " + dependency.getActualFilePath()
322                         + " - yarn.lock was identified if generated using synp the lock file may not be in the correct format.";
323                 LOGGER.error(msg);
324                 throw new AnalysisException("msg", ex);
325             }
326             LOGGER.error("NodeAuditAnalyzer failed on {}", dependency.getActualFilePath());
327             throw ex;
328         }
329     }

331     /**
332      * Analyzes the package and package-lock files by extracting dependency
333      * information, creating a payload to submit to the npm audit API,
334      * submitting the payload, and returning the identified advisories.
335      *
336      * @param file a reference to the package-lock.json
337      * @param dependency a reference to the dependency-object for the
338      * package-lock.json
339      * @param dependencyMap a collection of module/version pairs; during
340      * creation of the payload the dependency map is populated with the
341      * module/version information.
342      * @return a list of advisories
343      * @throws AnalysisException thrown when there is an error creating or
344      * submitting the npm audit API payload
345      */
346     private List<Advisory> legacyAnalysis(final File file, Dependency dependency, Map<String, String> dependencyMap)
347             throws AnalysisException {
348 
349         try (JsonReader jsonReader = Json.createReader(FileUtils.openInputStream(file))) {
350 
351             // Retrieves the contents of package-lock.json from the Dependency
352             final JsonObject packageJson = jsonReader.readObject();
353 
354             final String projectName = packageJson.getString("name", "");
355             final String projectVersion = packageJson.getString("version", "");
356             if (!projectName.isEmpty()) {
357                 dependency.setName(projectName);
358             }
359             if (!projectVersion.isEmpty()) {
360                 dependency.setVersion(projectVersion);
361             }
362 
363             // Modify the payload to meet the NPM Audit API requirements
364             final JsonObject payload = NpmPayloadBuilder.build(packageJson, dependencyMap);
365 
366             // Submits the package payload to the nsp check service
367             return searcher.submitPackage(payload);
368 
369         } catch (URLConnectionFailureException e) {
370             this.setEnabled(false);
371             throw new AnalysisException("Failed to connect to the NPM Audit API (NodeAuditAnalyzer); the analyzer "
372                     + "is being disabled and may result in false negatives.", e);
373         } catch (IOException e) {
374             LOGGER.debug("Error reading dependency or connecting to NPM Audit API", e);
375             this.setEnabled(false);
376             throw new AnalysisException("Failed to read results from the NPM Audit API (NodeAuditAnalyzer); "
377                     + "the analyzer is being disabled and may result in false negatives.", e);
378         } catch (JsonException e) {
379             throw new AnalysisException(String.format("Failed to parse %s file from the NPM Audit API "
380                     + "(NodeAuditAnalyzer).", file.getPath()), e);
381         } catch (SearchException ex) {
382             LOGGER.error("NodeAuditAnalyzer failed on {}", dependency.getActualFilePath());
383             throw ex;
384         }
385     }
