44       * Test of getEcosystem method, of class CveEcosystemMapper.
45       */
46      @Test
47      public void testGetEcosystem() {
48          CveEcosystemMapper mapper = new CveEcosystemMapper();
49          String value = "There is a vulnerability in some.java file";
50          assertEquals(JarAnalyzer.DEPENDENCY_ECOSYSTEM, mapper.getEcosystem(asCve(value)));
51          assertNull(mapper.getEcosystem(asCve(value,"cpe:2.3:a:apache:struts:1.2.1:*:*:*:*:*:*:*")));
52      }
53  
54      private DefCveItem asCve(String description, String... cpe) {
55          DefCveItem defCveItem = new DefCveItem();
56  
57          Description d = new Description();
58  
59          LangString string = new LangString();
60          string.setLang("en");
61          string.setValue(description);
62  
63          d.getDescriptionData().add(string);
64  
65          CVEJSON40Min11 cve = new CVEJSON40Min11();
66          cve.setDescription(d);
67  
68          defCveItem.setCve(cve);
69  
70          final DefConfigurations configurations = new DefConfigurations();
71          final DefNode node = new DefNode();
72          final List<DefCpeMatch> matches = new ArrayList<>();
73          final DefCpeMatch m = new DefCpeMatch();
74          m.setCpe23Uri("cpe:2.3:a:owasp:dependency-check:5.0.0:*:*:*:*:*:*:*");
75          matches.add(m);
76          for (String s : cpe) {
77              final DefCpeMatch cpeMatch = new DefCpeMatch();
78              cpeMatch.setCpe23Uri(s);
79              matches.add(cpeMatch);
80          }
81          node.setCpeMatch(matches);
82          List<DefNode> nodes = new ArrayList<>();
83          nodes.add(node);
84          configurations.setNodes(nodes);
85  
86          defCveItem.setConfigurations(configurations);
87  
88          return defCveItem;
89      }
